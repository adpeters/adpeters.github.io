from flask import current_app

import sqlalchemy as sa

from flask_cc.error_handlers import APIException
from flask_cc.utils import TemporaryOverrideConfig


db = SQLAlchemy()
migrate = Migrate()


def commit_safe():
    try:
        db.session.commit()
    except Exception as e:
        db.session.rollback()
        raise e


class CRUDMixin(object):
    def save(self):
        db.session.add(self)
        commit_safe()
        current_app.logger.info("Saved object {}".format(self))

    def delete(self):
        db.session.delete(self)
        commit_safe()
        current_app.logger.info("Deleted object {}".format(self))


class TimestampMixin(object):

[docs]def drop_tables():
    """Drop all tables in the database."""

    db.session.commit()
    db.session.remove()
    db.drop_all(bind=None)
    current_app.logger.info("All tables dropped.")




[docs]def create_tables(remove_existing_tables=True):
    """
    Create all tables in the database.

    :param bool remove_existing_tables: Option to drop all tables before creating new ones. This
                                        has the effect of cleaning up whatever exists in the
                                        database already.
    """

    if remove_existing_tables:
        drop_tables()
    db.create_all(bind=None)
    current_app.logger.info("All tables created.")




[docs]def query_lc_data(query, params=None, **execute_kwargs):
    """
    Raw query of database specified in "lc_data_web" SQLAlchemy bind.

    :param str query: Query to send to database.
    :param dict params: Parameters to send to SQLAlchemy execute method.
    :param kwargs execute_kwargs: Keyword arguments sent to SQLAlchemy execute method.
    :return: results of query from database
    :rtype: sqlalchemy.engine.ResultProxy
    """

    try:
        if params is None:
            params = {}
        return db.get_engine(current_app, 'lc_data_web').execute(sa.text(query), *params, **execute_kwargs)
    except sa.exc.OperationalError as e:
        current_app.logger.error(e)
        raise APIException("Error when trying to query database.")



class CustomDbCredentials(TemporaryOverrideConfig):
    def __init__(self, user, password):
        kwargs = dict(FLASK_DB_USER=user, FLASK_DB_PASSWORD=password)
        return super(CustomDbCredentials, self).__init__(**kwargs)

    def __enter__(self):
        app = super(CustomDbCredentials, self).__enter__()
        db.session.remove()
        return app

    def __exit__(self, exc_type, exc_value, exc_traceback):
        super(CustomDbCredentials, self).__exit__(exc_type, exc_value, exc_traceback)
        db.session.remove()
