[docs]def fill_vars(value):
    """
    Fills a variable specified in the json with a value coming from either the key store or environment variables.

    Notes:
    Tries to fill from key store first, then environment variables.

    :param str value: Variable name to be found in key store or environment variables.
    :rtype: str
    :returns: The value found for the given variable.
    """
    if not isinstance(value, basestring):
        try:
            return [fill_vars(i) for i in value]
        except Exception:
            return value

    pattern = r"\$\{([\w-]+)\}"
    env_vars = re.findall(pattern, value)
    if len(env_vars) > 0:
        for var in env_vars:
            new_value = None
            try:
                new_value = get_stored_value(var)
            except Exception:
                new_value = os.environ.get(var, None)
            if new_value is None:
                raise KeyError("Could not find {} in key vault or environment variables. This variable is necessary to load intitial data.".format(var))
            value = re.sub(r"\$\{" + var + r"\}", new_value, value)
    return value




[docs]def process_object(d):
    """Turns a JSON representation of an object into a Flask object."""
    otype = d["type"] + "Schema"
    if otype not in schemas.schema_list:
        current_app.logger.warn("Invalid object type: {}".format(otype))
        return None

    # get schema class for the object
    cls = getattr(schemas, otype)

    # fill in any environment variables in properties
    for prop, value in d["properties"].items():
        new_value = fill_vars(value)
        d["properties"][prop] = new_value

    result = cls().load(d["properties"], session=db.session)
    result = result.errors or result.data
    return result




[docs]def db_initial_setup(admin_user_password):
    """
    Loads objects to be created in the database on initial creation.

    Notes:
    - Objects are specified in json file db_initial_setup.json.
    - To specify an object to be loaded in development, test, or production only
      use an "environments" property. It can be a string or a list. Possible values are:
      * development
      * test
      * production
      * all
      Default value is all if not specified.
    """

    with open(os.path.join(current_app.root_path, "db_initial_setup.json"), "r") as f:
        jf = json.load(f)
    initial_objects = jf["initial_objects"]

    # set env variable for admin_user_password so it's properly filled in
    os.environ["FLASK_DEFAULT_USER_PASSWORD"] = admin_user_password

    try:
        for o in initial_objects:
            env = o.get("environment", ["all"])

            if isinstance(env, basestring):
                env = [env]

            do_save = False
            is_test = current_app.config.get("TESTING", False)
            for e in env:
                if e == "all":
                    do_save = True
                    break
                elif e == "test" and is_test:
                    do_save = True
                    break
                elif e == "development" and not is_test and current_app.config["ENV"] == "development":
                    do_save = True
                    break
                elif e == "production" and not is_test and current_app.config["ENV"] == "production":
                    do_save = True
                    break

            if do_save:
                orm_object = process_object(o)

                if orm_object is not None:
                    db.session.add(orm_object)
        db.session.commit()
        current_app.logger.info("Initial database objects created.")
    except Exception as e:
        db.session.rollback()
        raise e
