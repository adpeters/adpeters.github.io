[docs]class Config(object):
    """Base configuration values for app."""

    # flask config
    APP_NAME = 'index'
    DEBUG = False
    PERMANENT_SESSION_LIFETIME = datetime.timedelta(minutes=60)
    PROPAGATE_EXCEPTIONS = True
    SECRET_KEY = get_stored_value("FLASK-SECRET")
    SESSION_REFRESH_EACH_REQUEST = True

    # flask-login config
    REMEMBER_COOKIE_DURATION = datetime.timedelta(minutes=60)
    REMEMBER_COOKIE_REFRESH_EACH_REQUEST = True

    # flask-sqlalchemy config
    FLASK_DB_USER = os.environ.get("FLASK_DB_USER")
    FLASK_DB_PASSWORD = get_stored_value("FLASK-DB-PASSWORD")
    FLASK_DB_HOST = os.environ.get("FLASK_DB_HOST")
    FLASK_DB_PORT = os.environ.get("FLASK_DB_PORT")
    FLASK_DB_NAME = os.environ.get("FLASK_DB_NAME")
    LC_DB_USER = os.environ.get("LC_DB_USER")
    LC_DB_PASSWORD = get_stored_value("LC-DB-PASSWORD")
    LC_DB_HOST = os.environ.get("LC_DB_HOST")
    LC_DB_NAME = os.environ.get("LC_DB_NAME")
    SQLALCHEMY_TRACK_MODIFICATIONS = False

    # flask-mail config
    MAIL_SERVER = "smtp.sendgrid.net"
    MAIL_PORT = 587
    MAIL_USE_TLS = True
    MAIL_USERNAME = "apikey"
    MAIL_DEFAULT_SENDER = "no-reply@galeheaddev.com"
    MAIL_PASSWORD = get_stored_value("SENDGRID-API-KEY")

    @property
    def SQLALCHEMY_DATABASE_URI(self):
        return "postgresql://{}:{}@{}:{}/{}".format(
            self.FLASK_DB_USER,
            quote_plus(self.FLASK_DB_PASSWORD),
            self.FLASK_DB_HOST,
            self.FLASK_DB_PORT,
            self.FLASK_DB_NAME
        )

    @property
    def SQLALCHEMY_BINDS(self):
        return {
            "lc_data_web": "postgresql://{}:{}@{}/{}".format(
                self.LC_DB_USER,
                quote_plus(self.LC_DB_PASSWORD),
                self.LC_DB_HOST,
                self.LC_DB_NAME
            )
        }

    @property
    def DB_DOMAIN(self):
        try:
            return "@" + self.FLASK_DB_USER.split("@")[1]
        except Exception:
            return ""

    # custom config values
    GEOSERVER_URL_BASE = 'http://fenrir:8085/geoserver/'
    GEOSERVER_PATH = 'ows?version%3D1.3.0'
    GEOSERVER_SLOPE_PATH = 'www/slope/{z}/{x}/{y}.png'
    GEOSERVER_HEATMAP_PATH = 'www/heatmap/{z}/{x}/{y}.png'
    GEOSERVER_SOILS_PATH = 'www/soil/{z}/{x}/{y}.mvt'
