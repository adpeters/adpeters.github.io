def init_logging(app):
    formatter = UTCFormatter("%(asctime)s %(levelname)s %(name)s.%(module)s: %(message)s")
    handler = ConcurrentRotatingFileHandler(
        os.path.join(app.root_path, 'log.txt'),
        maxBytes=1024 * 1024)
    handler.setFormatter(formatter)
    logging.getLogger('waitress').setLevel(logging.DEBUG)
    logging.getLogger('waitress').addHandler(handler)
    app.logger.setLevel(logging.DEBUG)
    app.logger.addHandler(handler)

    # configure request logger
    request_handler = ConcurrentRotatingFileHandler(
        os.path.join(app.root_path, 'request_log.txt'),
        maxBytes=1024 * 1024)
    request_formatter = RequestFormatter("%(asctime)s %(levelname)s %(name)s: %(request)s %(message)s")
    request_handler.setFormatter(request_formatter)
    request_logger.setLevel(logging.DEBUG)
    request_logger.addHandler(request_handler)
