from azure.core.exceptions import ResourceNotFoundError
from azure.keyvault.secrets import SecretClient
from azure.identity import DefaultAzureCredential
import os


client = None


[docs]def init_vault_client():
    """Initialize the Key Vault client in azure_kv.client using credentials from environment variables."""

    credentials = DefaultAzureCredential()
    client = SecretClient(vault_url=os.environ.get("VAULT_URL"), credential=credentials)




[docs]def get_secret(secret_name, default_value=None):
    """
    Get a secret from the Azure Key Vault.

    :param str secret_name: Name of the secret to get the value for from KeyVault.
    :param str default_value: Value to return if secret is not found. If None,
                              will raise ResourceNotFoundError if value not found.
    :rtype: str
    :returns: Value of secret if found or default_value if not.
    """

    try:
        secret_bundle = client.get_secret(secret_name)
    except ResourceNotFoundError as e:
        if default_value is None:
            raise e
        return default_value
    return secret_bundle.value
