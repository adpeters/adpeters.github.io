def geojson_result(result, geom=True):
    if geom is False:
        return json.dumps({
            'type': 'FeatureCollection',
            'features': [
                {"type": "Feature",
                 "properties": OrderedDict(zip(result.keys(), row)),
                 } for row in result]
        })
    else:
        return json.dumps({
            'type': 'FeatureCollection',
            'features': [
                {"type": "Feature",
                 "geometry": json.loads(row[result.keys().index('geojson')]),
                 "properties": OrderedDict(zip(result.keys(), row)),
                 } for row in result]
        })

@api.route('/parcels_select', methods=['POST'])
@login_required
@gzipped
def parcels_select():
    x0, y0, x1, y1 = request.values.get('x0'), request.values.get('y0'), request.values.get('x1'), request.values.get('y1')

    if x0 is None or y0 is None or x1 is None or y1 is None:
        raise BadAPIRequest("Missing coordinates for parcel selection.", source=request.url)

    query = """
        select *
        from parcels2_view_query
        where geom && st_buffer(ST_SetSrid(st_centroid(ST_MakeEnvelope(:x0, :y0, :x1, :y1)), 3857), 11265)
        """

    try:
        return geojson_result(query_lc_data(query, x0=x0, y0=y0, x1=x1, y1=y1))
    except Exception as e:
        raise APIException(message=str(e), source=request.url)

@api.route('/parcels_buffer', methods=['POST'])
@login_required
def parcels_buffer():
    # print(request.args.get('filter'))
    if request.args.get('filter') == 'false':
        query = """
        select *
        from parcels2_view_query
        where geom && st_setsrid({buffer}::geometry, 3857)
        and st_intersects(geom, st_setsrid({buffer}::geometry, 3857))
        and usable_acres >= {usable}
        limit 1000
        """.format(
            buffer=request.get_data().replace('"', "'"),
            usable=request.args.get('usable')
        )
        return geojson_result(query_lc_data(query))
    elif request.args.get('filter') == 'true':
        query = """
        and geom && st_setsrid({buffer}::geometry, 3857)
        and st_intersects(geom, st_setsrid({buffer}::geometry, 3857))
        """.format(
            buffer=request.get_data().replace('"', "'"),
        )
        return query

def get_ipac_url(wkt):
    ipac_url = 'https://ecos.fws.gov/ipac/location'
    ipac_params = {
        "project.name": "Galehead Development",
        "requestor": "http://www.galeheaddev.com"
    }
    ipac_params["projectLocationWKT"] = wkt
    try:
        r = requests.post(ipac_url, params=ipac_params, timeout=10)
        return r.url
    except Exception as e:
        current_app.logger.error(e)
        raise APIException("Error generating IPAC report")

    """
    Generates a PDF report using Mapfish and return link to it.

    :param dict specs_data: Dictionary of data to be written to the specs file.
    :param str in_specs: Name of specs file.
    :param str in_config: Name of config file.
    :param str in_jrxml: Name of template file.

    :param dict extra_files: Dictionary of file_name: data pairs to be created for this report.
    :rtype: str
    :returns: URL to PDF report.

    if os.name == 'posix':
        print_script = os.path.join(current_app.root_path, 'core-3.18.2', 'bin', 'print')
        report_dir = os.path.join(current_app.root_path, 'tmp', 'reports')
    else:
        print_script = os.path.join(current_app.root_path, 'core-3.18.2', 'bin', 'print.bat')
        report_dir = os.path.join("C:\\Windows", "Temp", "reports")

    # generate new folder for report
    base_folder_name = os.path.join(current_app.root_path, 'api', 'reports', 'conf')
    image_folder = os.path.join(current_app.static_folder, 'images')
    folder_name, report_id = get_new_report_path(base_folder_name=report_dir)

    if 'CIA' not in in_config:
        # copy north arrow and pick and shovel image
        north_arrow = os.path.join(folder_name, 'north_arrow.png')
        shovel_logo = os.path.join(folder_name, 'Pick_and_shovel.png')
        shutil.copyfile(os.path.join(image_folder, 'north_arrow.png'), north_arrow)
        shutil.copyfile(os.path.join(image_folder, 'pick_and_shovel_watermark.png'), shovel_logo)

    config_file = os.path.join(folder_name, in_config)
    output_file = os.path.join(folder_name, output)
    jrxml_file = os.path.join(folder_name, os.path.split(in_jrxml)[1])
    grd_logo = os.path.join(folder_name, 'GRD_logo_kl.png')

    # copy config file
    shutil.copyfile(os.path.join(base_folder_name, in_config), config_file)

    # copy template
    shutil.copyfile(os.path.join(base_folder_name, in_jrxml), jrxml_file)

    # copy logo and fonts
    shutil.copyfile(os.path.join(image_folder, 'GRD_logo_kl.png'), grd_logo)

    # write the specs file
    spec_file = os.path.join(folder_name, in_specs)
    with open(spec_file, 'w') as out:
        out.write(json.dumps(specs_data))

    # write any other files
    if extra_files is not None:
        for file_name, data in extra_files.items():
            path = os.path.join(folder_name, file_name)
            with open(path, "w") as f:
                f.write(data)

    subprocess.check_output([
        print_script,
        '-verbose', '1',
        "-config", config_file,
        "-spec", spec_file,
        "-output", output_file
    ])
    return url_for("home.reports", report_id=report_id)



@api.route('/preCIA', methods=['GET', 'POST'])
@login_required
def preCIA():
    sql_pre_cia = preCIA_query()

    qstr_pre_cia = sql_pre_cia.format(
        table='parcels2_view_query',
        key=request.args.get('key'),
        state=request.args.get('state'),
    )

    q_pre_cia = query_lc_data(qstr_pre_cia)

    cia_ipac = get_ipac_url(request.get_data().replace('"', "'"))
    try:
        i = requests.get(cia_ipac)
        ipac_tag = re.findall(r'<span class="pull-right">\d+</span>', i.text)
        species_list = re.findall(r'"short-name">[a-z-\s]+<', i.text, flags=re.IGNORECASE)
        species_count = re.findall(r'\d+', ipac_tag[0])[0]
        if len(ipac_tag) == 2:
            crit_hab = 0
            bird_count = re.findall(r'\d+', ipac_tag[1])[0]
        elif len(ipac_tag) == 3:
            crit_hab = re.findall(r'\d+', ipac_tag[1])[0]
            bird_count = re.findall(r'\d+', ipac_tag[2])[0]
        else:
            crit_hab = 0
            bird_count = 0
        species_names = [s.split('>')[1].replace('<', '') for s in species_list]
        t_and_e_species = species_names[:int(species_count)]
    except Exception:
        species_count = '-1'
        t_and_e_species = 'unknown'
        crit_hab = -1
        bird_count = -1

    pre_cia_dict = {}
    pre_cia_dict['layout'] = '11x17 landscape'
    pre_cia_dict['outputFormat'] = 'pdf'

    for row in q_pre_cia:
        json_data = dict(zip(q_pre_cia.keys(), row))
    json_data['name'] = request.args.get('name')
    json_data['species_count'] = species_count
    if type(t_and_e_species) is list:
        json_data['species_names'] = ', '.join(t_and_e_species)
    else:
        json_data['species_names'] = t_and_e_species
    json_data['bird_count'] = bird_count
    json_data['crit_hab'] = crit_hab
    if not json_data['warnings']:
        if int(species_count) > 0:
            json_data['warnings'] = species_count + ' sensitive species'
        else:
            json_data['warnings'] = 'N/A'
    else:
        if int(species_count) > 0:
            warnings = json_data['warnings']
            json_data['warnings'] = warnings + species_count + ' sensitive species'.replace(',,', '').replace(', ,', '')

    # fill in any missing values so mapfish doesn't error
    for key, value in json_data.items():
        if value is None:
            json_data[key] = 'None'

    pre_cia_dict['attributes'] = json_data

    gen_rprt = get_report_url(specs_data=pre_cia_dict, in_specs='requestData_preCIA.json', in_config='config_preCIA.yaml',
                              in_jrxml='pre_cia.jrxml',
                              output='output_preCIA.pdf')
    return jsonify({"url": gen_rprt})


@api.route('/map', methods=['GET', 'POST'])
def map():
    map_key = request.args.get('map_key')
    map_state = request.args.get('map_state')
    map_name = request.args.get('name')
    map_center = request.args.get('center')

@api.route('/reverse_geocode', methods=['GET', 'POST'])
def reverse_geocode():
    coords = request.args.get('coords')
    geocode_url = (
        'https://reverse.geocoder.api.here.com/6.2/reversegeocode.json?app_id=lHF3VB1kJnlbF0dRh8Pk'
        '&app_code=5r5YSxhnwI1MRQeK0AkTJw&mode=retrieveAddresses&prox={coords}'.format(coords=coords)
    )
    geocode_resp = requests.get(geocode_url)
    geocode_content = json.loads(geocode_resp.text)
    geocode_data = geocode_content['Response']['View'][0]['Result']
    distances = []
    address_summary = ''
    address_city = ''
    address_state = ''
    address_zip = ''

    try:
        for g in geocode_data:
            if g['MatchLevel'] == 'houseNumber':
                distances.append(g['Distance'])
        closest_addr = min(distances)
        for gg in geocode_data:
            if gg['MatchLevel'] == 'houseNumber' and gg['Distance'] == closest_addr:
                address_summary += gg['Location']['Address']['HouseNumber'] + ' ' + gg['Location']['Address']['Street']
                address_city += gg['Location']['Address']['City']
                address_state += gg['Location']['Address']['State']
                address_zip += gg['Location']['Address']['PostalCode']
    except Exception:
        pass
    return json.dumps([address_summary, address_city, address_state, address_zip])


@api.route("/stop_query", methods=['POST'])
@login_required
def stop_query():
    if request.method == 'POST':
        data = request.get_json()
        if data is None:
            return ("No JSON payload", 400)
        filter_clause = data.get('queryString')
        stop_query_string = parcel_filter_query(filter_clause)
        # print stop_query_string
        select_statement = "SELECT datname,pid,state,query FROM pg_stat_activity where query = '" + stop_query_string.replace(
            "'", "''") + "'"
        # print select_statement
        kill_q = query_lc_data(select_statement)
        result_dicts = [dict(zip(row.keys(), row)) for row in kill_q]
        # print result_dicts
        if len(result_dicts) == 0:
            return ("No such running query found", 404)
        result_pid = result_dicts[0]['pid']
        kill_statement = 'SELECT pg_terminate_backend({});'.format(result_pid)
        # print kill_statement
        query_lc_data(kill_statement)

        return jsonify({
            "status": "success",
            "message": "Query terminated"
        })
