def jwt_encode(obj):
    token = jwt.encode(obj, key=current_app.config["SECRET_KEY"], algorithm="HS256")
    return token.decode("utf-8")


def jwt_decode(token):
    return jwt.decode(token, key=current_app.config["SECRET_KEY"], algorithms=["HS256"])


def is_safe_url(url):
    if url is None or len(url) == 0:
        return True
    parsed_host = urlparse(request.host_url)
    parsed_url = urlparse(url)
    if len(parsed_url.netloc) > 0 and parsed_url.netloc != parsed_host.netloc:
        current_app.logger.warning("Invalid next parameter: {}".format(url))
        return False
    if not parsed_url.path.startswith("/"):
        current_app.logger.warning("Invalid next parameter: {}".format(url))
        return False
    return True
