    )

@anonymous_user_required
@auth.route("/login", methods=["GET", "POST"])
def login():
    form = LoginForm()

    if form.validate_on_submit():
        # Successfully logged in, We can now access the saved user object
        # via form.user.
        user = User.query.filter_active_by(email=form.email.data).first()
        if user is None or not user.verify_password(form.password.data):
            flash('Invalid email or password')
            return redirect(url_for('auth.login'))

        login_user(user, remember=True)
        session.permanent = True
        next = request.args.get('next')
        if not is_safe_url(next):
            return redirect(url_for("home.index"))

        return redirect(next or url_for('home.index'))

    return render_template("auth/login.html", form=form)

def profile_view(form_class, user, title="", is_profile=False):
    is_admin_user = current_user.has_any_of_roles("Admin")
    form = form_class(obj=user)
    if is_admin_user:
        form.roles.choices = [(role.id, role.name) for role in Role.query.all()]
    delete_form = DeleteUserForm(obj=user)

    if delete_form.validate_on_submit() and "delete" in request.form:
        # verify user being deleted matches id of user in form
        if user.id == int(request.form["id"]):
            user.delete()
            return redirect(url_for("auth.manage_users"))
    elif form.validate_on_submit():
        form_valid = True
        # existing password must be entered in order to change email or password
        if is_profile and user.email != form.email.data and not form.existing_password.data:
            form.email.errors.append(ValidationError("You must enter your password in order to change your email."))
            form_valid = False

        user.first_name = form.first_name.data
        user.last_name = form.last_name.data
        if is_admin_user:
            user.roles = [Role.query.get(r) for r in form.roles.data]
        password_updated = False
        email_changed = False

        if form_valid:
            try:
                # if existing password was entered, ensure it is correct, or else log
                # the user out and send them to login page
                if form.existing_password.data and not user.verify_password(form.existing_password.data):
                    flash("Invalid password. Please login again.")
                    logout_user()
                    return redirect(url_for("auth.login"))
                elif form.existing_password.data:
                    if form.password.data:
                        user.password = form.password.data
                        password_updated = True
                    if user.email != form.email.data:
                        email_changed = True
                        user.email = form.email.data
                        user.email_confirmed = False
                        # update the id as well so user is logged out from any current sessions
                        user.set_alternate_id()
            except AttributeError as e:
                if "'EditUserAdminForm' object has no attribute 'existing_password'" in e:
                    # this is the case when an Admin is editing another user's profile
                    # in this case, entering passwords is not an option.
                    pass
                else:
                    raise e

            flash("Profile successfully updated.")
            user.save()

            if password_updated:
                flash("Password successfully updated, please login with your new password.")
            if email_changed:
                send_email_confirmation_message(user)
                flash("Email successfully changed, please check your email for a confirmation link.")
            if password_updated or email_changed:
                return redirect(url_for("auth.login"))
    elif is_admin_user:
        form.roles.data = [role.id for role in user.roles]

    return render_template("auth/profile.html", form=form, delete_form=delete_form, title=title, is_profile=is_profile)


@auth.route("/reauthenticate", methods=["GET", "POST"])
@login_required
def reauthenticate():
    form = LoginForm()

    if form.validate_on_submit():
        user = User.query.filter_active_by(email=form.email.data).first()
        if user.id != current_user.id or user is None or not user.verify_password(form.password.data):
            logout_user()
            flash('Invalid email or password')
            return redirect(url_for('auth.login'))

        login_user(user, remember=True)

        next = request.args.get('next')
        if not is_safe_url(next):
            return redirect(url_for("home.index"))

        return redirect(next or url_for('home.index'))

    return render_template("auth/login.html", form=form)


@auth.route("/register", methods=["GET", "POST"])
@fresh_login_required
@roles_accepted("Admin")
def register():
    form = RegistrationForm()

    is_admin_user = current_user.has_any_of_roles("Admin")
    # add role options to form
    if is_admin_user:
        form.roles.choices = [(role.id, role.name) for role in Role.query.all()]

    if form.validate_on_submit():
        user = User(email=form.email.data,
                    password=form.password.data,
                    first_name=form.first_name.data,
                    last_name=form.last_name.data)
        if is_admin_user:
            user.roles = [Role.query.get(r) for r in form.roles.data]
        user.save()
        # send message to confirm email
        send_email_confirmation_message(user)
        flash("User successfully registered. Tell them to check their email to activate their account.")
        return redirect(url_for("auth.manage_users"))
    return render_template("auth/register.html", form=form)

        return redirect(url_for("auth.login"))

    return render_template("auth/reset_password_request.html", form=form)


@auth.route("/users", methods=["GET", "POST"])
@login_required
@roles_accepted("Admin")
def manage_users():
    users = User.query.all()
    forms = []
    for user in users:
        prefix = "form-{}".format(user.id)
        form = UserListForm(obj=user, prefix=prefix)
        forms.append(form)

    submitted_form = None
    if request.method == "POST":
        # find the form that was submitted
        for form in forms:
            if "{}{}".format(form._prefix, "id") in request.form:
                submitted_form = form
                break

    if submitted_form is not None:
        # set default return message
        message = "Unable to change user status."
        try:
            if submitted_form.validate():
                user = User.query.get(submitted_form.id.data)
                if user.disabled != submitted_form.disabled.data:
                    user.disabled = submitted_form.disabled.data
                    user.save()
                    new_status = "disabled" if user.disabled else "enabled"
                    message = "{} successfully {}.".format(user.full_name, new_status)
            else:
                current_app.logger.warning(submitted_form.errors)
                raise APIException("Unable to change user status.")
        except Exception as e:
            current_app.logger.error(e.message)
            raise APIException("Unable to change user status.")

        return jsonify({"status": "success", "message": message})

    return render_template("auth/manage_users.html", forms=forms)
